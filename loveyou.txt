// CONTADOR

const startDate=new Date("2025-08-22");

function updateCounter(){

let now=new Date();

let diff=now-startDate;

let days=Math.floor(diff/(1000*60*60*24));

document.getElementById("counter").innerHTML=
`❤️ Llevamos ${days} días juntos ❤️`;

}
